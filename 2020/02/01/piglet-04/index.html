  • case { type: “case”, test: AST, consequent: [ …AST ] }
  • while { type: “while”, test: AST, consequent: [ …AST ] }
  • do … while { type: “do-while”, test: AST, consequent: [ …AST ] }

  • continue { type: “continue” }
